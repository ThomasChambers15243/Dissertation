112    def tokenize_code(self, file: str) -> tuple[dict,dict,int,int]:
113        """
114        Tokenises the code in the file as compatible Halstead operands and operators
115        :param file: File Path
116        :return: operator and operand counts and distinct operators and operands dictionary
117        """
118        # Gets each line in the file and stores tokens in a list
119        self.file = file
120        with open(self.file, 'r', encoding="utf8") as file:
121            for line in file:
122                # Adds 5 spaces to the end of the line to prevent index out of range errors
123                self.line = r"".join(line) + "     "
124                self.current = 0
125
126                # Loop through each line
127                while self.current <= len(self.line) - 3:
128                    # Handles comments and Blocks
129                    if not self.in_string:
130                        if self.handle_comments():
131                            break
132                        if self.is_quote_block_chars() and self.handle_block_quote():
133                            continue
134                        if self.in_quote_block:
135                            self.current += 1
136                            continue
137
138                    # Checks String
139                    if self.in_string_block():
140                        self.add_string()
141
142                    # Skip Null Tokens
143                    if self.skip_null_tokens():
144                        continue
145
146                    # Checks Letter
147                    if self.line[self.current].isalpha() or self.line[self.current] == '_':
148                        self.add_operand()
149
150                    # Checks Operator
151                    if self.is_operator_char():
152                        self.add_operator()
153
154                    # Digit/number
155                    if self.line[self.current].isdigit():
156                        self.add_number()
157
158        return self.distinct_operators, self.distinct_operands, self.operator_count, self.operand_count

160    def handle_comments(self) -> bool:
161        """
162        Resets the current index to 0 if the current line is a comment
163        :return: Bool
164        """
165        if self.line[self.current] == self.comment_char and not self.in_quote_block:
166            self.current = 0
167            return True
168        return False

170    def handle_block_quote(self) -> bool:
171        """
172        Skips block quote chars and flips inQuoteBlock flag
173        :return: Bool
174        """
175        self.current += 3
176        self.in_quote_block = not self.in_quote_block
177        return self.in_quote_block

179    def is_quote_block_chars(self) -> bool:
180        """
181        Uses a lookahead to check if the current token is a block quote
182        :return: bool
183        """
184        is_quote = self.line[self.current]
185        for i in range(1, 3):
186            is_quote += self.line[self.current + i]
187        return is_quote in self.block_quote_chars_three

189    def skip_null_tokens(self) -> bool:
190        """
191        Skips null tokens
192        :return: bool
193        """
194        while self.is_null_token():
195            self.current += 1
196            if self.current >= len(self.line) - 2:
197                return True
198        return False

200    def in_string_block(self) -> bool:
201        """
202        Flips inString flag if token is a string character
203        :return: bool
204        """
205        if self.line[self.current] in self.string_chars:
206            self.in_string = not self.in_string
207        return self.in_string

209    def add_string(self) -> None:
210        """
211        Gets string and adds it to the token list
212        :return: None
213        """
214        string = ""
215        self.current += 1
216        # Loops until end of string character is found. Does not add "|' to string
217        while self.in_string_block():
218            string += self.line[self.current]
219            self.current += 1
220            # Handles escape characters
221            if self.line[self.current] == '\\':
222                string += self.line[self.current + 1]
223                self.current += 2
224        # Adds token and updates current to skip " character
225        self.add_token(string)
226        self.current += 1
